function test (int i, int j, float v, float x, float[100] a) {
    while (true) {
        do i = i + 1; while (a[i] < v);
        do j = j - 1; while (a[j] > v);
        if (i >= j) break;
        x = a[i]; a[i] = a[j]; a[j] = x;
    }
}

function main () {
   int i, j;
   float v, x;
   float[100] a;
   test(i, j, v, x, a);
}
